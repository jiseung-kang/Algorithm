# 적록색약

## 문제 이해
단지번호붙이기와 비슷한 문제. R, G, B에 대한 확인을 추가한다.

|생각한 구현 방법|설명|
|-|-|
|BFS|R/G/B를 구분하는 배열과 RG/B를 구분하는 배열을 따로 만들어 너비우선탐색을 진행한다.|

## 코드
```
N = int(input())
div = [[] for _ in range(N)]
for i in range(N):
  tmp = input()
  for j in range(N):
    div[i].append(tmp[j])

# 이동할 좌표
dx = [-1,1,0,0]
dy = [0,0,-1,1]

def bfs(a,b,d):
  queue = [[a,b]]
  while queue:
    x, y = queue[0][0], queue[0][1]
    del queue[0]
    # 현재 색깔을 저장
    cur = d[x][y]
    # 탐색처리
    d[x][y] = ''
    for i in range(4):
      nx = x + dx[i]
      ny = y + dy[i]
      # 같은 색깔이면 너비우선탐색 queue에 추가한다.
      if nx >= 0 and nx < N and ny >= 0 and ny < N:
        if d[nx][ny] == cur and d[nx][ny]!='':
          queue.append([nx, ny])

RGdiv = [[] for _ in range(N)]
# RG를 구분하지 못하는 배열
for i in range(N):
  for j in range(N):
    if div[i][j] == 'G':
      RGdiv[i].append('R')
    else:
      RGdiv[i].append(div[i][j])

rb = 0
rgb = 0
# RGB를 구분하는 배열
for i in range(N):
  for j in range(N):
    if RGdiv[i][j] != '':
      bfs(i,j,RGdiv)
      rb += 1
    if div[i][j] != '':
      bfs(i,j,div)
      rgb += 1

print(rgb, rb)
```
