# 단지번호붙이기

## 문제 이해
너비우선탐색으로 연결되어 있는 위치들의 개수를 센다.

|생각한 구현 방법|설명|
|-|-|
|BFS|연결된 위치들을 끝까지 탐색하며 개수를 센다.|

## 코드
```
n = int(input())
apartment = [[] for i in range(n)]
# 이동할 좌표
dx = [-1,1,0,0]
dy = [0,0,-1,1]

for i in range(n):
  tmp = input()
  for j in range(n):
    apartment[i].append(int(tmp[j]))

def bfs(a, b):
  queue = [[a,b]]
  # 아파트 개수를 초기화한다.
  num = 0
  while queue:
    x, y = queue[0][0], queue[0][1]
    del queue[0]
    for i in range(4):
      nx = x + dx[i]
      ny = y + dy[i]
      if nx >= 0 and nx < n and ny >= 0 and ny < n:
        if apartment[nx][ny] == 1:
          # 연결된 아파트를 추가한다.
          num += 1
          # 탐색된 아파트는 -1로 갱신해서 더 탐색하지 않는다.
          apartment[nx][ny] = -1
          queue.append([nx, ny])
  return num

ans = []

for i in range(n):
  for j in range(n):
    if apartment[i][j] == 1:
      tmp = bfs(i,j)
      # 연결된 단지가 없으면 아파트개수는 1개
      if tmp == 0:
        ans.append(1)
      else:
        ans.append(tmp)

# 정렬해서 단지개수와 단지별 아파트 개수 출력
ans.sort()
length = len(ans)
print(len(ans))
for a in ans:
  print(a)
```
